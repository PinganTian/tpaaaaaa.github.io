HDU - 3715 Go Deeper

Posted on 2019-09-03 | In 图论

题目大意：

给你一个递推式，问你最多能循环几层？

go(int dep, int n, int m)
begin
output the value of dep.
if dep < m and x[a[dep]] + x[b[dep]] != c[dep] then go(dep + 1, n, m)
end

分析：

同样的 二分答案+ 2-SAT。 代码不难敲，重要的是思路。

代码：

二分部分：

int L = 0;
int R = m;
int mid;
int ans = -1;
while(L <= R) //一般整型就可以这样，浮点形的等于不好用。
{
    mid = (L + R) / 2;
    built(mid);             //一直到第mid个约束条件，那么深度就是mid - 1；每次都要建一个新图哦！！！
    if(solve())             //如果满足约束条件，说明深度可以变大，添加更多的约束条件
    {
        ans = max(ans, mid);
        L = mid + 1;        //如果mid = L的话  可以无限循环
    }
    else                    //不满足，说明减少一些深度
        R = mid - 1;
}

完整代码+分析：

/*
 Date              : 2019-09-03-18.47.52
 Question_Number   ：hdu 3715
 Question_Name     ：2 sat
 Description       :
 n = 2, m = 1;
 a 0
 b 1
 c 0

 x[ai] + x[bi] != ci;
 a1 表示a选0   意思是x这个数组里面ai位置选0
 a2 表示a选1                              1
 因为x这个数组长为n
 所以图中实际上有2n个点
 这2n个点，编号从0开始

 0 1   2 3   4 5

 a1  为 ai * 2;
 a2  为 ai * 2 + 1;
 b1 为

 i*2  i*2+1
 得到约束条件
 a1 - > b2;
 b1 - > a2;

 x[a] + x[b] != 1;
 a2 > b2;
 b2 > a2;
 a1 > b1;
 b1 > a1;

 x[a] + x[b] != 2;
 a2 > b1;
 b2 > a1;
 // 要使得他们不为0

 //为什么要二分呢？？  因为可以简化深度越大，意味着所连的边就越多
                                        //就越有可能ai 和 ai^1 形成一个环
                                        // 假如一个图中  所有 ai 和 ai ^ 1 都不在一个环内
                                        // 那么删掉一些边，这些对点肯定也不再一个环内
                                        // 这就出现了单调性
                                                //  最多有m个条件
        //  所以可以二分边界   0 <= dep <= m
*/
#include<functional>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<string>
#include<vector>
#include<stack>
#include<bitset>
#include<cstdlib>
#include<cmath>
#include<set>
#include<list>
#include<deque>
#include<map>
#include<queue>
using namespace std;
typedef long long ll;
const double PI = acos(-1.0);
const double eps = 1e-6;
const int INF = 0x3f3f3f3f;
const int MAXN = 2*200 + 10;  //图中点的数目为
const int MAXM = 1e4 + 10;
int a[MAXM], b[MAXM], c[MAXM];
int dfn[MAXN], low[MAXN], belong[MAXN], head[MAXN];
bool vis[MAXN];
int n, m, cnt, sum, tot;
stack < int > st;
struct Edge
{
    int v, next;
};
Edge edge[MAXN * MAXN];
void addEdge(int u, int v)
{
    cnt ++;
    edge[cnt].v = v;
    edge[cnt].next = head[u];
    head[u] = cnt;
}
void init()
{
    memset(head, -1, sizeof head);
    memset(dfn, 0, sizeof dfn);
    memset(vis, false, sizeof vis);
    cnt = sum = tot = 0;
}
void built(int mid)
{
    init();                         //建图之前  清空图
    for(int i = 1; i <= mid; i++)
    {
        int a1 = a[i] * 2;
        int a2 = a[i] * 2 + 1;
        int b1 = b[i] * 2;
        int b2 = b[i] * 2 + 1;
        switch(c[i])
        {
                case 0:
                    {
                        addEdge(a1, b2);
                        addEdge(b1, a2);
                        break;
                    }
                case 1:
                    {
                        addEdge(a2, b2);
                        addEdge(b2, a2);
                        addEdge(a1, b1);
                        addEdge(b1, a1);
                        break;
                    }
                case 2:
                    {
                        addEdge(a2, b1);
                        addEdge(b2, a1);
                        break;
                    }
        }
    }

}
void Tarjan(int u)
{
    vis[u] = true;
    st.push(u);
    low[u] = dfn[u] = ++ tot;
    for(int i = head[u]; i != -1; i = edge[i].next)
    {
        int v = edge[i].v;
        if(dfn[v] == 0)
        {
            Tarjan(v);
            low[u] = min(low[u], low[v]);
        }
        else if(vis[v])
            low[u] = min(low[u], dfn[v]);
    }
    if(dfn[u] == low[u])
    {
        sum ++;
        while(st.top() != u)
        {
            int temp = st.top();
            st.pop();
            vis[temp] = false;
            belong[temp] = sum;
        }
        st.pop();
        vis[u] = false;
        belong[u] = sum;
    }
}
bool solve()
{
    for(int i = 0; i < 2*n; i++)       //图中点的编号为0 ... 2n - 1
        if(dfn[i] == 0)
            Tarjan(i);
    for(int i = 0; i < 2*n; i+= 2)
        if(belong[i] == belong[i + 1])
            return false;

    return true;
}
int main()
{
    int t;
    cin >>t;
    while(t--)
    {
        scanf("%d%d", &n, &m);
        for(int i = 1; i <= m; i++)
            scanf("%d%d%d", &a[i], &b[i], &c[i]);
        int L = 0;
        int R = m;
        int mid;
        int ans = -1;
        while(L <= R)
        {
            mid = (L + R) / 2;
            built(mid);             //一直到第mid个约束条件，那么深度就是mid - 1；
            if(solve())             //如果满足约束条件，说明深度可以变大，添加更多的约束条件
            {
                ans = max(ans, mid);
                L = mid + 1;        //如果mid = L的话  可以无限循环
            }
            else                    //不满足，说明减少一些深度
                R = mid - 1;
        }

         printf("%d\n", ans);
    }
    return 0;
}
