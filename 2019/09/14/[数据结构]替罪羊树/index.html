[数据结构]替罪羊树

Posted on 2019-09-14 | In 数据结构

替罪羊树是一棵不需要旋转操作的AVL树，因此比较简单，适合入门学习。

视频：https://www.bilibili.com/video/av58948184/?spm_id_from=333.788.videocard.0

模板题：https://www.luogu.org/problem/P3369

树的节点 Node

struct Node
{
    int l, r, val;
// l,r为子节点的下标，val为该节点的权值
    int siz, fac;
// siz为该子树的大小，fac为该子树在删除操作之后的大小
    bool exist;
// 该点是否存在
};

新节点的构建

//用val构建一个新的节点，now为他的地址
void newnode(int &now, int val)
{
    now = ++cnt;
    tzy[now].val = val;
    tzy[now].siz = tzy[now].fac = 1;
    tzy[now].exist = true;
}

添加节点的操作——add

//在树中插入一个val的值
void add(int &now, int val)
{
    if(now == 0)
    {
        newnode(now, val);//插入完成
        check(root, now); //从根节点开始往下找，看看是不是存在不平衡的点
        return;
    }
    tzy[now].siz++;
    tzy[now].fac++;

    if(val < tzy[now].val)
        add(tzy[now].l, val);
    else
        add(tzy[now].r, val);
}

删除操作 —— del

//在树中删除权值为val的点,惰性删除
void del(int now, int val)
{

    if(tzy[now].exist && tzy[now].val == val)
    {
        tzy[now].exist = false;
        tzy[now].fac --;
        check(root, now);
        return;
    }
    tzy[now].fac --;
    if(val < tzy[now].val)
        del(tzy[now].l, val);
    else
        del(tzy[now].r, val);
}

检查树是否需要重构——check

void check(int &now, int ori)
{
    if(now == ori)
        return;
        //该点不平衡,将其重构，更新
    if(imbalance(now))
    {
        rebuilt(now);
        update(root, now);
        return;
    }
    if(tzy[ori].val < tzy[now].val)
        check(tzy[now].l, ori);
    else
        check(tzy[now].r, ori);
}

判断某个点是否不平衡——imbalance

bool imbalance(int now)
{
    if(max(tzy[tzy[now].l].siz, tzy[tzy[now].r].siz ) > tzy[now].siz * alpha
       || tzy[now].siz - tzy[now].fac > tzy[now].siz * 0.3 )
        return true;
    return false;
}

重构函数——rebuilt

void rebuilt(int &now)
{
    vec.clear();
    ldr(now);
    if(vec.empty())
    {
        now = 0;
        return;
    }
    lift(0, vec.size() - 1, now);
}

其中的中序遍历

void ldr(int now)
{
    if(!now) return;
    ldr(tzy[now].l);
    if(tzy[now].exist)
        vec.push_back(now);
    ldr(tzy[now].r);
}

以及lift函数

void lift(int l, int r, int &now)
{
    if(l == r)
    {
        now = vec[l];
        tzy[now].l = tzy[now].r = 0;
        tzy[now].siz = tzy[now].fac = 1;
        return;
    }
    int mid = md(l, r);
    while(l < mid && tzy[vec[mid]].val == tzy[vec[mid - 1]].val)
        mid --;
    now = vec[mid];
    if(l < mid)
        lift(l, mid - 1, tzy[now].l);
    else
        tzy[now].l = 0;

    lift(mid + 1, r, tzy[now].r);
    tzy[now].siz = tzy[tzy[now].l].siz + tzy[tzy[now].r].siz + 1;
    tzy[now].fac = tzy[tzy[now].l].fac + tzy[tzy[now].r].fac + 1;

}

祖父节点的siz更新

void update(int now, int ori)
{
    if(!now)
        return;
    if(tzy[ori].val < tzy[now].val)
        update(tzy[now].l, ori);
    else
        update(tzy[now].r, ori);

    tzy[now].siz = tzy[tzy[now].l].siz + tzy[tzy[now].r].siz + 1;

}

查询值的排名——getrank

int getrank(int val)
{
    int now = root;
    int rankk = 1;
    while(now)
    {
        if(val <= tzy[now].val)
            now = tzy[now].l;
        else
        {
            rankk += tzy[now].exist + tzy[tzy[now].l].fac;
            now = tzy[now].r;
        }
    }
    return rankk;
}

某个排名的值查询——getnum

int getnum(int rak)
{
    int now = root;
    while(now)
    {
        if(tzy[tzy[now].l].fac + tzy[now].exist == rak)
            break;
        else if(tzy[tzy[now].l].fac >= rak)
            now = tzy[now].l;
        else
        {
            rak -= tzy[tzy[now].l].fac + tzy[now].exist;
            now = tzy[now].r;
        }
    }
    return tzy[now].val;
}

获得某个数的前驱和后继

前驱：getnum(getrank(x) - 1)
后继：getnum(getrank(x + 1))

完整的板子：

//AC的板子
#include <iostream>
#include <vector>
#include <cstdio>
#define md(x, y)  (x + y) / 2
using namespace std;
const int MAXN = 1e5 + 10;
int cnt = 0, root;
vector < int > vec;
const double alpha = 0.75;
struct Node
{
    int l, r, val;
    int siz, fac;
    bool exist;
};
Node tzy[MAXN];
void newnode(int &now, int val)
{
    now = ++cnt;
    tzy[now].l = tzy[now].r = 0;
    tzy[now].val = val;
    tzy[now].siz = tzy[now].fac = 1;
    tzy[now].exist = true;
}
bool imbalance(int now)
{
    if(max(tzy[tzy[now].l].siz, tzy[tzy[now].r].siz ) > tzy[now].siz * alpha
       || tzy[now].siz - tzy[now].fac > tzy[now].siz * 0.3 )
        return true;
        return false;
}
//将now为根的子树重建，首先用中序遍历
void ldr(int now)
{
    if(!now)
        return;
    ldr(tzy[now].l);
    if(tzy[now].exist)
        vec.push_back(now);
    ldr(tzy[now].r);
}
void lift(int l, int r, int &now)
{
    if(l == r)
    {
        now = vec[l];
        tzy[now].l = tzy[now].r = 0;
        tzy[now].siz = tzy[now].fac = 1;
        return;
    }
    int mid = md(l, r);
    while(l < mid && tzy[vec[mid]].val == tzy[vec[mid - 1]].val)
        mid --;
    now = vec[mid];
    if(l < mid)
        lift(l, mid - 1, tzy[now].l);
    else
        tzy[now].l = 0;
    lift(mid + 1, r, tzy[now].r);
    tzy[now].siz = tzy[tzy[now].l].siz + tzy[tzy[now].r].siz + 1;
    tzy[now].fac = tzy[tzy[now].l].fac + tzy[tzy[now].r].fac + 1;

}
int getrank(int val)
{
    int now = root;
    int rankk = 1;
    while(now)
    {
        if(val <= tzy[now].val)
            now = tzy[now].l;
        else
        {
            rankk += tzy[now].exist + tzy[tzy[now].l].fac;
            now = tzy[now].r;
        }
    }
    return rankk;
}
int getnum(int rak)
{
    int now = root;
    while(now)

        else
        {
            rak -= tzy[tzy[now].l].fac + tzy[now].exist;
            now = tzy[now].r;
        }

    }
    return tzy[now].val;
}
void update(int now, int ori)
{
    if(!now)
        return;
    if(tzy[ori].val < tzy[now].val)
        update(tzy[now].l, ori);
    else
        update(tzy[now].r, ori);

    tzy[now].siz = tzy[tzy[now].l].siz + tzy[tzy[now].r].siz + 1;

}
void rebuilt(int &now)
{
    vec.clear();
    ldr(now);
    if(vec.empty())
    {
        now = 0;
        return;
    }
    lift(0, vec.size() - 1, now);
}
void check(int &now, int ori)
{
    if(now == ori)
        return;
        //该点不平衡,将其重构，更新
    if(imbalance(now))
    {
        rebuilt(now);
        update(root, now);
        return;
    }
    if(tzy[ori].val < tzy[now].val)
        check(tzy[now].l, ori);
    else
        check(tzy[now].r, ori);
}
//在now这个位置插入val
void del(int now, int val)
{
    if(tzy[now].exist&&tzy[now].val == val)
    {
        tzy[now].exist = false;
        tzy[now].fac --;
        check(root, now);

    if(val < tzy[now].val)
        del(tzy[now].l, val);
    else
        del(tzy[now].r, val);
}
void add(int &now, int val)
{

        newnode(now, val);//插入完成

}
