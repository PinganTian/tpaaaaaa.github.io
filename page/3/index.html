void solve(int x)
{
    int temp = x;
    int s = 0;
    while(temp <= MAXN)             //往左移.
    {
        cnt[temp]++;
        step[temp] += s;
        temp <<= 1;
        s++;
    }
    s = 1;
    temp = x;
    while(temp > 1)
    {
        if(temp & 1)
        {
            int cc = temp;
            cc >>= 1;
            int ss = 0;
            while(cc <= MAXN)
            {
                cnt[cc]++;
                step[cc] += s + ss;
                ss++;
                cc <<= 1;
            }
            temp >>= 1;
            s++;
        }
        else
        {
            temp >>= 1;
            cnt[temp]++;
            step[temp] += s;
            s++;
        }

    }

}

{
    while(scanf("%d", &n) != EOF)
    {

        memset(step, 0, sizeof step);
        for(int i = 1; i <= n; i++)
        {
            int temp;
            scanf("%d", &temp);
            solve(temp);
        }
        for(int i = 1; i <= MAXN; i++)
            if(cnt[i] == n)
                ans = min(ans, step[i]);
